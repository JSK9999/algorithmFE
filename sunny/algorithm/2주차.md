function solution(n) {
  var answer = 0;
  const ternaryArr = [];
  
  const returnToTernary = (n) => {
    if(n >= 3) {
      ternaryArr.unshift(n%3);
      n = Math.floor(n/3);
      returnToTernary(n);
    } else {
      return ternaryArr.unshift(n)
    }
  }
  returnToTernary(n);
  
  let index = 0;
  const returnToDecimal = (arr) => {
    if(!arr[index] && arr[index] !== 0){
      return answer;
    } else {
      answer += (arr[index] * Math.pow(3, index))
      index++;
      returnToDecimal(arr);
    }
  }
  
  returnToDecimal(ternaryArr)
  return answer;
}
