```
function solution(n) {
 var answer = 0;
 const ternaryArr = [];

  const returnToTernary = (n) => {
    if (n >= 3) {
      ternaryArr.push(n%3);
      n = Math.floor(n/3);

      returnToTernary(n);
    } else {
      return ternaryArr.push(n)
    }
  }

  returnToTernary(n);

  answer = parseInt(ternaryArr.join(''), 3);
  return answer;
}
```

```
function solution(string) {
  let answer = [];

  […string].forEach((str) => {
    if (!answer.includes(str)) answer.push(str);
  })

  return answer.join('');
}
```
